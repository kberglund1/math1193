\title{Vertex Finder}
\author{Kenneth Berglund}

\begin{document}
\begin{abstract}
    Finding a parabola with a given vertex.
\end{abstract}
\maketitle

\begin{enumerate}
    \item Write down the equation of a parabola whose graph has
    its vertex at \((-1, 3)\).
    \item Can you find another?
    \item How many such parabolas are there?
\end{enumerate}
