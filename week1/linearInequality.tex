\documentclass[noauthor,noinstructornotes]{ximera}

\title{Linear Inequality}
\author{Math 1193 Design Team}

\begin{document}
\begin{abstract}
    Solving and graphing a linear inequality.
\end{abstract}
\maketitle

Solve the inequality \(-2x + 4 \ge 0\). Shade the 
solution set on the number line, then write the 
solution set in interval notation.

\begin{instructorNotes}
This is meant to see how students go about solving inequalities. 
Do they perform the operations in the correct order? Do they know 
about flipping the inequality? Also, it makes sure students understand 
intervals and interval notation.
\end{instructorNotes}

\end{document}
